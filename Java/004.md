# 4. Median of Two Sorted Arrays

## Solution 1 (O(log(m+n)))

```java
// Binary search
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int total_length = nums1.length + nums2.length;
        int t = total_length / 2;
        if (total_length % 2 == 0) {
            return (findK(nums1, nums2, t - 1) + findK(nums1, nums2, t)) / 2.0;
        } else {
            return findK(nums1, nums2, t);
        }
    }

    public double findK(int[] nums1, int[] nums2, int k) {
        if (nums1.length == 0)
            return nums2[k];
        if (nums2.length == 0)
            return nums1[k];
        if (k == 0)
            return Math.min(nums1[0], nums2[0]);

        int p1 = nums1.length / 2;
        int p2 = nums2.length / 2;
        if (nums1[p1] > nums2[p2]) {
            if (k > p1 + p2) {
                return findK(nums1, Arrays.copyOfRange(
                		nums2, p2 + 1, nums2.length), k - p2 - 1);
            } else {
                return findK(Arrays.copyOfRange(nums1, 0, p1), nums2, k);
            }
        } else {
            if (k > p1 + p2) {
                return findK(Arrays.copyOfRange(nums1, p1 + 1, nums1.length),
                		nums2, k - p1 - 1);
            } else {
                return findK(nums1, Arrays.copyOfRange(nums2, 0, p2), k);
            }
        }
    }
}
```
