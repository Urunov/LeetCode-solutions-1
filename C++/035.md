# 35. Search Insert Position

## Solution 1 (O(log(n)))

```cpp
// Binary search
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;
        while (left <= right) {
            int mid = (left + right) / 2;
            if (target == nums[mid]) {
                return mid;
            } else if (target < nums[mid] && (mid == 0 || target > nums[mid - 1])) {
                return mid;
            } else if (target > nums[mid] && mid == nums.size() - 1) {
                return mid + 1;
            } else if (target > nums[mid]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return -1;
    }
};
```
