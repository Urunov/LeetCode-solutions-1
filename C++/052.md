# 52. N-Queens II

## Solution 1

```cpp
// DFS
class Solution {
public:
    int ans = 0;
    int totalNQueens(int n) {
        vector<string> cur;
        _solve(n, 0, cur);
        return ans;
    }

    void _solve(int n, int row, vector<string>& cur) {
        if (row == n) {
            ans += 1;
            return;
        }
        for (int col = 0; col < n; col++) {
            string t = "";
            for (int i = 0; i < n; i++) {
                if (i == col)
                    t += 'Q';
                else
                    t += '.';
            }
            cur.push_back(t);
            if(_isValid(cur))
                _solve(n, row + 1, cur);
            cur.pop_back();
        }
    }

    bool _isValid(vector<string>& cur) {
        int col = cur.size() - 1;
        int ind = (int)cur[col].find('Q');
        for (int i = 0; i < col; i++) {
            int diff = abs((int)cur[i].find('Q') - ind);
            if (diff == 0 || diff == col - i)
                return false;
        }
        return true;
    }
};
```
